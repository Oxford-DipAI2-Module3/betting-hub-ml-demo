    <script>

        const MODELS = {{ models|tojson }};

        const USER_PROFILES = {{ user_profiles|tojson }};


        let currentTiles = [];

        let currentModel = 'lightgbm';

        let currentUserType = 'casual';


        // Initialize

        displayModelInfo(MODELS[currentModel]);

        displayUserInfo(USER_PROFILES[currentUserType]);


        document.getElementById('fetchBtn').addEventListener('click', fetchTiles);

        document.getElementById('userSelect').addEventListener('change', (e) => {

            currentUserType = e.target.value;

            displayUserInfo(USER_PROFILES[currentUserType]);

            if (currentTiles.length > 0) {

                rankTiles(currentTiles);

            }

        });


        async function fetchTiles() {

            const btn = document.getElementById('fetchBtn');

            const status = document.getElementById('status');

            

            btn.disabled = true;

            status.textContent = '🔄 Fetching live tiles from Checkd API...';

            

            try {

                const res = await fetch('/api/fetch-tiles', {

                    method: 'POST',

                    headers: {'Content-Type': 'application/json'}

                });

                const data = await res.json();

                

                if (!data.success) {

                    status.textContent = `❌ Error: ${data.error}`;

                    return;

                }

                

                currentTiles = data.tiles;

                status.textContent = `✅ Fetched ${data.tiles.length} tiles. Applying ML...`;

                

                displayCurrent(data.tiles);

                await rankTiles(data.tiles);

                

                status.textContent = `✅ Complete! Showing ${data.tiles.length} tiles`;

            } catch (e) {

                status.textContent = `❌ Error: ${e.message}`;

            } finally {

                btn.disabled = false;

            }

        }


        function displayUserInfo(profile) {

            const info = document.getElementById('userInfo');

            info.innerHTML = `<strong>${profile.name}:</strong> ${profile.strategy}`;

        }


        function displayCurrent(tiles) {

            const feed = document.getElementById('currentFeed');

            feed.innerHTML = tiles.map(t => createTile(t, t.position, false)).join('');

        }


        function createTile(tile, pos, showScore = false, score = null, change = null) {

            const kw = {};

            if (tile.keywords) {

                tile.keywords.forEach(k => kw[k.name] = k.value);

            }


            let cls = '';

            let changeHtml = '';

            if (change !== null) {

                if (change > 0) {

                    cls = 'up';

                    changeHtml = `<div class="tile-change up">📈 UP ${change}</div>`;

                } else if (change < 0) {

                    cls = 'down';

                    changeHtml = `<div class="tile-change down">📉 DOWN ${Math.abs(change)}</div>`;

                }

            }


            const scoreHtml = showScore && score ? `<div class="tile-score">ML Score: ${score.toFixed(4)}</div>` : '';


            return `

                <div class="tile ${cls}">

                    <div class="tile-pos">${pos}</div>

                    <div class="tile-desc">${tile.description || 'N/A'}</div>

                    <div class="tile-meta">

                        <span>📊 ${kw.bet_type || 'N/A'}</span>

                        <span>🏢 ${kw.bookmaker || 'N/A'}</span>

                        <span>⚽ ${kw.sport || 'N/A'}</span>

                    </div>

                    ${scoreHtml}

                    ${changeHtml}

                </div>

            `;

        }


        async function rankTiles(tiles) {

            const res = await fetch('/api/rank-tiles', {

                method: 'POST',

                headers: {'Content-Type': 'application/json'},

                body: JSON.stringify({tiles, model: currentModel, user_type: currentUserType})

            });

            const data = await res.json();

            

            if (!data.success) {

                document.getElementById('status').textContent = `❌ ${data.error}`;

                return;

            }

            

            displayML(data.tiles);

            displayModelInfo(data.model);

            

            const personalizationNote = data.user_profile.sessions > 0 ? 

                '<span style="color: #16a34a;">✓ PERSONALIZED</span>' : 

                '<span style="color: #f59e0b;">⚠ COLD START</span>';

            document.getElementById('mlModelName').innerHTML = `${data.model.name} ${personalizationNote}`;

        }


        function displayML(scored) {

            const feed = document.getElementById('mlFeed');

            feed.innerHTML = scored.map(s => 

                createTile(s.tile, s.recommended_position, true, s.score, s.change)

            ).join('');

        }


        function displayModelInfo(model) {

            document.getElementById('metricAUC').textContent = model.auc_roc ? model.auc_roc.toFixed(4) : 'N/A';

            document.getElementById('metricActualCTR').textContent = model.actual_ctr || 'N/A';

            document.getElementById('metricPredictedCTR').textContent = model.predicted_ctr || 'N/A';

            document.getElementById('metricPeriod').textContent = model.training_period || 'N/A';

            document.getElementById('personalizationInfo').textContent = model.personalization || 'N/A';

            

            const prosEl = document.getElementById('pros');

            prosEl.innerHTML = model.pros.map(p => `<li>${p}</li>`).join('');

            

            const consEl = document.getElementById('cons');

            consEl.innerHTML = model.cons.map(c => `<li>${c}</li>`).join('');

        }
